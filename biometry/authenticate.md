```cpp
#include "UltimateIOSToolkit/Biometry.h"

void UMyClass::MyAuthenticate()
{
	UIT::Biometry::Authenticate
	(
		/* Reason */
		TEXT("We want to make sure it is you"),
		
		/* Fallback Title */
		TEXT("Continue with Pin Code"),
		
		/* Cancel Title */
		TEXT("Use Pin Code Instead"),
		
		/* Callback */
		FOnBiometricAuthentication::CreateLambda([](bool bAuthenticated) -> void
		{
			if (bAuthenticated)
			{
				// User passed the biometric authentication.
			}
			
			else
			{
				// Biometric authentication cancelled or failed.
			}
		})
	);
}
```

```cpp
#include "UltimateIOSToolkit/Biometry.h"

void UMyClass::CheckBiometry()
{
	switch (UIT::Biometry::GetBiometryType())
	{
	case EIOSBiometryType::None:    /* Biometry is not available. */ break;
	case EIOSBiometryType::FaceID:  /* The device has FaceID.     */ break;
	case EIOSBiometryType::TouchID: /* The device has TouchID.    */ break;
	}
}
```

```cpp
#include "UltimateIOSToolkit/UI.h"

void UMyClass::SendMail()
{
	UIT::UI::SendMail
	(
		// The email recipients (to).
		{ TEXT("john.doe@domain.com") },
		
		// The email CCs.
		{ TEXT("clark.kent@superheros.org") },
		
		// The email BCCs.
		{ TEXT("peter.parker@superheros.org") },
		
		// The email's subject.
		TEXT("The email subject."),
		
		// The email's body.
		TEXT("This is the body."),
		
		// If the body supports HTML or not.
		true,
		
		// An optional callback called when the overlay is dismissed.
		FOnMailSent::CreateLambda([](EIOSMailComposeResult Result, EIOSMailComposeErrorCode Error) -> void
		{
			switch (Result)
			{
			// The user canceled the operation.
			case EIOSMailComposeResult::Cancelled: break;
		
			// The email message was saved in the user's drafts folder.
			case EIOSMailComposeResult::Saved: break;
		
			// The email message was queued in the user's outbox.
			case EIOSMailComposeResult::Sent: break;
		
			// The email message was not saved or queued, possibly due to an error.
			case EIOSMailComposeResult::Failed: break;
			}
		})
	);
}
```

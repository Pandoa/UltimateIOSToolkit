#include "UltimateIOSToolkit/UI.h"

void UMyClass::ShowDefinition()
{
	const FString Term = TEXT("Game Engine");
	
	if (UIT::UI::DictionaryHasDefinition(Term))
	{
		UIT::UI::ShowDefinition(Term);
	}
	else
	{
		// No definition for the term.
	}
}

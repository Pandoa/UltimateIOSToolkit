```cpp
#include "UltimateIOSToolkit/UI.h"

void UMyClass::ShowPicker()
{
	UIT::UI::ShowColorPicker
	(
		/* Default color picked. */
		FLinearColor::White, 
		
		/* If alpha is supported. */
		true,
		
		/* The callback executed when the user picked a color. */
		FOnColorPicked::CreateLambda([](const FLinearColor& Color) -> void
		{
			/* Do something with Color. */
		})
	);
}
```
